    ```sql
      id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
      issue_number int NOT NULL,
      year int NOT NULL,
      issue_supplement int NOT NULL DEFAULT 0,
      date date,
      title text,
      source_url text UNIQUE,
      created_at timestamptz DEFAULT now(),
      UNIQUE(issue_number, year, issue_supplement)
    );

    CREATE TABLE IF NOT EXISTS dv_acts (
      id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
      issue_id uuid REFERENCES dv_issues(id),
      issue_number int NOT NULL,
      year int NOT NULL,
      act_number text,
      title text NOT NULL,
      act_type text,
      full_text text,
      source_url text UNIQUE,
      razdel int,
      summary_ai text,
      summary_ai_generated_at timestamptz,
      created_at timestamptz DEFAULT now()
    );

    -- 2. tsvector + GIN (simple config, weights A title / B act_type / C full_text capped at 50000 chars):
    ALTER TABLE dv_acts ADD COLUMN IF NOT EXISTS search_vector tsvector
      GENERATED ALWAYS AS (
        setweight(to_tsvector('simple', coalesce(title, '')), 'A') ||
        setweight(to_tsvector('simple', coalesce(act_type, '')), 'B') ||
        setweight(to_tsvector('simple', left(coalesce(full_text, ''), 50000)), 'C')
      ) STORED;

    CREATE INDEX IF NOT EXISTS dv_acts_fts ON dv_acts USING gin(search_vector);

    ALTER TABLE dv_issues ADD COLUMN IF NOT EXISTS search_vector tsvector
      GENERATED ALWAYS AS (
        setweight(to_tsvector('simple', coalesce(title, '')), 'A')
      ) STORED;

    CREATE INDEX IF NOT EXISTS dv_issues_fts ON dv_issues USING gin(search_vector);

    -- 3. Ranking RPC (mirrors Phase 2's intel_search_top blend 0.7 ts_rank + 0.3 recency_decay, half-life 365 days):
    CREATE OR REPLACE FUNCTION dv_search_top(
      q text,
      filter_year int DEFAULT NULL,
      filter_act_type text DEFAULT NULL,
      filter_from_date date DEFAULT NULL,
      filter_to_date date DEFAULT NULL,
      filter_from_issue int DEFAULT NULL,
      filter_to_issue int DEFAULT NULL,
      limit_n int DEFAULT 50
    ) RETURNS TABLE (
      id uuid,
      issue_id uuid,
      issue_number int,
      year int,
      date date,
      title text,
      act_type text,
      source_url text,
      lex real,
      rec real,
      score real
    ) LANGUAGE sql STABLE AS $$
      WITH q_ts AS (
        SELECT websearch_to_tsquery('simple', q) AS query
      )
      SELECT
        a.id,
        a.issue_id,
        a.issue_number,
        a.year,
        i.date,
        a.title,
        a.act_type,
        a.source_url,
        ts_rank(a.search_vector, q_ts.query) AS lex,
        exp(-EXTRACT(EPOCH FROM (now() - i.date::timestamptz)) / (365.0 * 86400)) AS rec,
        (0.7 * ts_rank(a.search_vector, q_ts.query)
           + 0.3 * exp(-EXTRACT(EPOCH FROM (now() - i.date::timestamptz)) / (365.0 * 86400))
        ) AS score
      FROM dv_acts a
      JOIN dv_issues i ON i.id = a.issue_id
      CROSS JOIN q_ts
      WHERE length(trim(q)) > 0
        AND a.search_vector @@ q_ts.query
        AND (filter_year IS NULL OR a.year = filter_year)
        AND (filter_act_type IS NULL OR a.act_type = filter_act_type)
        AND (filter_from_date IS NULL OR i.date >= filter_from_date)
        AND (filter_to_date IS NULL OR i.date <= filter_to_date)
        AND (filter_from_issue IS NULL OR a.issue_number >= filter_from_issue)
        AND (filter_to_issue IS NULL OR a.issue_number <= filter_to_issue)
      ORDER BY score DESC
      LIMIT limit_n;
    $$;
    ```

    ```typescript
    import { readFileSync } from "node:fs";
    import { Client } from "pg";

    function maskUrl(url: string): string {
      return url.replace(/(:[^:@/]+@)/, ":***@");
    }

    async function main() {
      const dbUrl = process.env.DATABASE_URL;
      if (!dbUrl) {
        console.error("Error: DATABASE_URL not set in environment.");
        console.error("Hint: source ../lex-brain/.env or supply DATABASE_URL inline.");
        process.exit(1);
      }

      const sqlPath = `${process.cwd()}/db/dv_schema.sql`;
      const sql = readFileSync(sqlPath, "utf-8");

      console.log(`Applying DV schema migration from ${sqlPath} to ${maskUrl(dbUrl)}…`);

      const client = new Client({ connectionString: dbUrl });
      await client.connect();

      try {
        await client.query("BEGIN");
        await client.query(sql);
        await client.query("COMMIT");
        console.log("Schema applied successfully.");
      } catch (err) {
        await client.query("ROLLBACK");
        console.error("Schema apply failed:", err);
        await client.end();
        process.exit(1);
      }

      // Probes
      let allOk = true;
      const probes: { name: string; query: string; expect: number }[] = [
        {
          name: "search_vector column on dv_acts",
          query: `SELECT column_name FROM information_schema.columns WHERE table_name='dv_acts' AND column_name='search_vector'`,
          expect: 1,
        },
        {
          name: "search_vector column on dv_issues",
          query: `SELECT column_name FROM information_schema.columns WHERE table_name='dv_issues' AND column_name='search_vector'`,
          expect: 1,
        },
        {
          name: "GIN indexes (dv_acts_fts + dv_issues_fts)",
          query: `SELECT indexname FROM pg_indexes WHERE indexname IN ('dv_acts_fts', 'dv_issues_fts')`,
          expect: 2,
        },
        {
          name: "dv_search_top RPC",
          query: `SELECT proname FROM pg_proc WHERE proname='dv_search_top'`,
          expect: 1,
        },
      ];

      for (const probe of probes) {
        const res = await client.query(probe.query);
        if (res.rowCount === probe.expect) {
          console.log(`OK: ${probe.name} (${res.rowCount}/${probe.expect})`);
        } else {
          console.error(`MISSING: ${probe.name} (got ${res.rowCount}, expected ${probe.expect})`);
          allOk = false;
        }
      }

      // Smoke: call the RPC with empty input — must NOT error
      try {
        const smoke = await client.query("SELECT * FROM dv_search_top('тест') LIMIT 1");
        console.log(`OK: dv_search_top('тест') returned ${smoke.rowCount} row(s).`);
      } catch (err) {
        console.error(`FAIL: dv_search_top smoke query failed:`, err);
        allOk = false;
      }

      await client.end();
      process.exit(allOk ? 0 : 1);
    }

    main().catch((err) => {
      console.error(err);
      process.exit(1);
    });
    ```

    ```python
    """JSF state extraction + polite scraping helpers for dv.parliament.bg.

    Phase 8 sibling to scripts/_lib/http_retry.py. http_retry.py is append-only per
    Phase 1 D-12: this module adds new helpers, does NOT modify the existing ones.

    The dv.parliament.bg site is Apache MyFaces + JSP. Stateful via jsessionid cookie
    + javax.faces.ViewState hidden input. See 08-RESEARCH.md §"Architecture" for the
    full navigation graph.
    """

    from __future__ import annotations

    import random
    import re
    import time
    from typing import Dict, Optional

    import httpx

    USER_AGENT = "lex-brain-scraper/1.0 (+https://lex-web-eta.vercel.app; non-commercial public-interest project)"

    _VIEW_STATE_RE = re.compile(
        r'name="javax\.faces\.ViewState"[^>]*value="([^"]+)"'
    )

    _OAM_SUBMIT_4ARG_RE = re.compile(
        r"""oamSubmitForm\(
            \s*'([^']+)'\s*,        # form name
            \s*'([^']+)'\s*,        # source id
            \s*null\s*,
            \s*\[(.*?)\]            # params array
            \s*\)""",
        re.VERBOSE | re.DOTALL,
    )

    _OAM_SUBMIT_3ARG_RE = re.compile(
        r"""oamSubmitForm\(
            \s*'([^']+)'\s*,        # form name
            \s*'([^']+)'\s*          # source id (no params)
            \)""",
        re.VERBOSE,
    )

    _OAM_PARAM_RE = re.compile(r"\['([^']+)'\s*,\s*'([^']*)'\]")

    _JSESSIONID_RE = re.compile(r";jsessionid=[A-F0-9]+", re.IGNORECASE)

    _ACT_TYPE_RE = re.compile(
        r"^(Закон|Наредба|Постановление|Указ|Решение|Обявление|Разпореждане|Заповед|Инструкция|Правилник)\b",
        re.UNICODE,
    )

    _VIEW_EXPIRED_RE = re.compile(
        r"(View has expired|ViewExpiredException)",
        re.IGNORECASE,
    )


    def extract_view_state(html: str) -> str:
        """Extract the javax.faces.ViewState value from a JSF response HTML.

        Raises ValueError if not found.
        """
        match = _VIEW_STATE_RE.search(html)
        if not match:
            raise ValueError("javax.faces.ViewState not found in response")
        return match.group(1)


    def parse_oam_submit(onclick: str) -> Dict[str, object]:
        """Parse an oamSubmitForm(...) call from an onclick attribute.

        Returns {"form": str, "source": str, "params": dict[str, str]}.
        Handles both 4-arg (with params) and 3-arg (no params) variants.

        Raises ValueError if the onclick doesn't contain a recognizable oamSubmitForm call.
        """
        m4 = _OAM_SUBMIT_4ARG_RE.search(onclick)
        if m4:
            form, source, params_blob = m4.group(1), m4.group(2), m4.group(3)
            params = {k: v for k, v in _OAM_PARAM_RE.findall(params_blob)}
            return {"form": form, "source": source, "params": params}

        m3 = _OAM_SUBMIT_3ARG_RE.search(onclick)
        if m3:
            form, source = m3.group(1), m3.group(2)
            return {"form": form, "source": source, "params": {}}

        raise ValueError(f"No oamSubmitForm() call found in: {onclick[:120]}…")


    def strip_jsessionid(url: str) -> str:
        """Remove the ;jsessionid=... segment from a URL.

        Session IDs are session-bound and useless outside the originating scrape session.
        Always strip before saving to dv_acts.source_url or dv_issues.source_url.
        """
        return _JSESSIONID_RE.sub("", url)


    def infer_act_type(title: str) -> str:
        """Infer the act type from the title's prefix word.

        Returns one of the 10 known Bulgarian legal-act types, or 'Other' if no match.
        Source: RESEARCH Q3 — verified against issue 2026/42 TOC.
        """
        match = _ACT_TYPE_RE.match(title.strip())
        return match.group(1) if match else "Other"


    def polite_sleep(min_s: float = 1.0, max_s: float = 2.0) -> None:
        """Sleep a random duration between min_s and max_s seconds.

        Default range (1.0–2.0 s) implements CONTEXT D-02's polite delay
        (1.5 s base + ±500 ms jitter).
        """
        time.sleep(random.uniform(min_s, max_s))


    def make_session(timeout_s: float = 30.0) -> httpx.Client:
        """Create an httpx.Client configured for dv.parliament.bg scraping.

        Per CONTEXT D-03: identifying User-Agent, follow redirects, cookie jar enabled.
        """
        return httpx.Client(
            follow_redirects=True,
            timeout=timeout_s,
            headers={"User-Agent": USER_AGENT},
        )


    def is_view_expired(response_text: str) -> bool:
        """Check if a JSF response indicates the session ViewState has expired.

        Tomcat default session timeout is 30 minutes idle (RESEARCH Q9). When this
        returns True, the scraper must re-establish a fresh session via make_session()
        + GET index.faces, then resume at the last completed (year, broi_).
        """
        return bool(_VIEW_EXPIRED_RE.search(response_text))
    ```

<threat_model>

**T-DV-01-01 Cache poison / partial-data persistence (Information disclosure)**
- Threat: scraper crashes mid-act, leaves `full_text=NULL` on a partial act; future reader sees incomplete content
- Mitigation: write `full_text` ONLY after successful GET; resumability scan refetches `WHERE full_text IS NULL OR length(full_text)=0`
- Severity: low (no security impact; data-quality only)

**T-DV-01-02 Government-site rate-limit retaliation (Reputational + Availability)**
- Threat: dv.parliament.bg implements aggressive rate limiting; our scraper IP gets blocked
- Mitigation: ≥1.5 s polite delay (D-02), jittered to avoid request correlation; identifying UA so the operator can contact us (D-03); honor 429 Retry-After + 503 backoff; abort on 403 with explicit user notification
- Severity: medium (reputational — we'd be a "bad actor" if hammering)

**T-DV-01-03 jsessionid leak in stored URLs (Tracking / session-bound data leakage)**
- Threat: ;jsessionid=... in source_url shared with users; URL contains a session token that, if still valid, could grant access to OUR scraper's session
- Mitigation: `strip_jsessionid()` applied to EVERY source_url before DB write (D-05); tested in test_dv_jsf.py; SQL probe in smoke-test confirms 0 leaks
- Severity: medium (session token leak; bounded by Tomcat's 30-min idle timeout)

**T-DV-01-04 Schema change breaks lex-brain (Tampering / Availability)**
- Threat: lex-brain depends on the shared Supabase; a destructive DDL would break ingestion
- Mitigation: SQL is purely additive (`IF NOT EXISTS` / `OR REPLACE`); no DROP, no ALTER COLUMN TYPE, no data movement; idempotent re-run is no-op; BLOCKING checkpoint (Task 3) requires human acknowledgement
- Severity: low (additive only; same pattern as Phase 2's intel_fts which already shipped without incident)

**T-DV-01-05 ViewState replay attack (Spoofing — against US, not the gov site)**
- Threat: An attacker observes our scraper's outbound traffic and replays our ViewState to inject responses
- Mitigation: TLS (HTTPS) end-to-end; ViewState is server-issued and bound to OUR session cookie; an attacker without the session cookie can't forge a valid ViewState
- Severity: low (TLS-protected; not a practical attack)

**T-DV-01-06 IMMUTABLE function-volatility regression (Tampering — against generated columns)**
- Threat: same shape as Phase 2's `array_to_string` STABLE-not-IMMUTABLE bug
- Mitigation: Phase 8 schema uses ONLY IMMUTABLE built-ins (`to_tsvector`, `coalesce`, `setweight`, `left`, `||`) — no STABLE-returning function inside the GENERATED expression; verified during research; no wrapper needed (RESEARCH Pitfall 5)
- Severity: low (verified absent)
