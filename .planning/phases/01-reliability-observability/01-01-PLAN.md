    ```python
    @contextlib.contextmanager
    def fetch_with_retry_stream(
        client: httpx.Client,
        url: str,
        **kwargs: Any,
    ):
        """Streaming sibling of fetch_with_retry_sync (D-12).

        Yields the streaming httpx.Response inside a context manager. Caller MUST
        iterate inside the ``with`` block — the connection closes on __exit__.

        Retries happen ONLY at the connection-establish boundary (before any
        bytes are consumed). Once the body starts streaming we never retry,
        because httpx raises StreamConsumed on second iteration of a partially
        consumed response (RESEARCH Pitfall 2).

        Same retry policy as fetch_with_retry_sync (D-13): _BACKOFF curve,
        _is_transient_status set, _log_* telemetry. Yields None on exhaustion
        (matches the helper convention from D-04).
        """
        total = len(_BACKOFF)
        for attempt in range(total + 1):
            try:
                cm = client.stream("GET", url, **kwargs)
                r = cm.__enter__()
            except (httpx.TimeoutException, httpx.TransportError) as e:
                if attempt == total:
                    _log_transport_giving_up(e, url)
                    yield None
                    return
                delay = _BACKOFF[attempt]
                _log_retry(attempt + 1, total, delay, type(e).__name__, url)
                time.sleep(delay)
                continue

            if _is_transient_status(r.status_code):
                cm.__exit__(None, None, None)
                if attempt == total:
                    _log_giving_up(str(r.status_code), url)
                    yield None
                    return
                delay = _BACKOFF[attempt]
                _log_retry(attempt + 1, total, delay, f"HTTP {r.status_code}", url)
                time.sleep(delay)
                continue

            try:
                yield r
            finally:
                cm.__exit__(None, None, None)
            return
    ```

    class _IterBytesAdapter(io.RawIOBase):
        """Minimal RawIOBase wrapper around an Iterator[bytes] so TextIOWrapper
        can read from it. Used to feed httpx Response.iter_bytes() chunks into
        csv.DictReader through TextIOWrapper(..., newline='').

        Per RESEARCH Pitfall 1: TextIOWrapper expects a buffer with .readinto(),
        not a bare iterator — this adapter bridges the gap.
        """
        def __init__(self, it):
            self._it = it
            self._buf = b""

        def readable(self) -> bool:
            return True

        def readinto(self, b) -> int:
            while not self._buf:
                try:
                    self._buf = next(self._it)
                except StopIteration:
                    return 0
            n = min(len(b), len(self._buf))
            b[:n] = self._buf[:n]
            self._buf = self._buf[n:]
            return n
    ```

    Locked decisions exercised:
    - D-14: peak RSS via psutil (NOT tracemalloc / memray — they miss
      C-layer buffering; D-14 explicitly chose OS-level RSS).
    - D-15: synthetic CSV generated in-test, deterministic, offline.
    - D-16: real-feed (live OpenSanctions ~300-500 MB CSV) is UAT, not CI.

    Set OPENSANCTIONS_TEST_MB env var to override fixture size (default 100).
    Floor is 100 MB so a buffered implementation (the OLD code) would clearly
    bust the 200 MB budget — Pitfall 6 in RESEARCH.
    """
    from __future__ import annotations

    import csv
    import io
    import os
    import sys
    import threading
    from pathlib import Path

    import httpx
    import psutil
    import pytest

    ROOT = Path(__file__).resolve().parent.parent
    sys.path.insert(0, str(ROOT))

    from scripts._lib.http_retry import fetch_with_retry_stream  # noqa: E402


    OPENSANCTIONS_HEADER = [
        "id", "schema", "name", "aliases", "birth_date", "countries",
        "addresses", "identifiers", "sanctions", "phones", "emails",
        "dataset", "first_seen", "last_seen",
    ]

    FIXTURE_MB = int(os.environ.get("OPENSANCTIONS_TEST_MB", "100"))


    class _IterBytesAdapter(io.RawIOBase):
        """Mirrors the adapter in scrape_opensanctions.py — kept local to the
        test so this file is self-contained and the test still exercises the
        same TextIOWrapper(..., newline='') path the production code uses."""
        def __init__(self, it):
            self._it = it
            self._buf = b""

        def readable(self) -> bool:
            return True

        def readinto(self, b) -> int:
            while not self._buf:
                try:
                    self._buf = next(self._it)
                except StopIteration:
                    return 0
            n = min(len(b), len(self._buf))
            b[:n] = self._buf[:n]
            self._buf = self._buf[n:]
            return n


    class PeakRssSampler:
        """Background thread polling psutil.Process().memory_info().rss every
        50 ms and tracking the max. Per D-14 / RESEARCH Pattern 2."""

        def __init__(self, interval_s: float = 0.05) -> None:
            self._interval = interval_s
            self._proc = psutil.Process()
            self._stop = threading.Event()
            self._thread: threading.Thread | None = None
            self.peak_bytes = 0

        def _loop(self) -> None:
            while not self._stop.is_set():
                rss = self._proc.memory_info().rss
                if rss > self.peak_bytes:
                    self.peak_bytes = rss
                self._stop.wait(self._interval)

        def start(self) -> None:
            self.peak_bytes = self._proc.memory_info().rss  # baseline
            self._thread = threading.Thread(target=self._loop, daemon=True)
            self._thread.start()

        def stop(self) -> None:
            self._stop.set()
            if self._thread:
                self._thread.join(timeout=1.0)

        @property
        def peak_mb(self) -> float:
            return self.peak_bytes / (1024 * 1024)


    def _write_synthetic_csv(path: Path, target_mb: int) -> None:
        """Write a deterministic CSV until file size >= target_mb. Mix BG/non-BG
        rows so the country-skip path is exercised. Includes a row with embedded
        newlines in a quoted field (D-15: validates newline='' handling)."""
        target_bytes = target_mb * 1024 * 1024
        with path.open("w", encoding="utf-8", newline="") as f:
            w = csv.writer(f)
            w.writerow(OPENSANCTIONS_HEADER)
            i = 0
            while path.stat().st_size < target_bytes:
                i += 1
                countries = "BG" if i % 7 == 0 else "RU"
                name = f"Test Entity {i}"
                if i % 1000 == 0:
                    # Embedded newline: validates that csv handles its own line
                    # splitting via newline="" — would break with iter_lines().
                    name = f"Multi\nLine\nName {i}"
                w.writerow([
                    f"opensanc-{i:08d}", "Person", name, "", "1980-01-01",
                    countries, "", "", "OFAC SDN", "", "", "default",
                    "2024-01-01", "2026-01-01",
                ])


    def _make_streaming_mock_client(csv_path: Path) -> httpx.Client:
        """httpx.MockTransport that streams the fixture from disk in 64 KiB
        chunks. CRITICAL: do NOT use content=csv_path.read_bytes() — that
        loads the fixture into memory and defeats the test (RESEARCH caveat
        in §httpx MockTransport for offline test)."""
        def handler(request: httpx.Request) -> httpx.Response:
            def gen():
                with csv_path.open("rb") as f:
                    while True:
                        chunk = f.read(65536)
                        if not chunk:
                            break
                        yield chunk
            return httpx.Response(
                200,
                headers={"content-type": "text/csv; charset=utf-8"},
                content=gen(),
            )
        return httpx.Client(transport=httpx.MockTransport(handler))


    @pytest.fixture(scope="module")
    def synthetic_csv(tmp_path_factory):
        p = tmp_path_factory.mktemp("os-fixture") / "synthetic.csv"
        _write_synthetic_csv(p, target_mb=FIXTURE_MB)
        size_mb = p.stat().st_size / (1024 * 1024)
        assert size_mb >= FIXTURE_MB - 1, (
            f"fixture only wrote {size_mb:.1f} MB; expected >={FIXTURE_MB}"
        )
        return p


    def test_peak_rss_under_200_mb(synthetic_csv, monkeypatch):
        """The OS-01 acceptance test: streamed parse against a >=100 MB CSV
        keeps peak RSS under 200 MB on the test runner."""
        monkeypatch.setattr("time.sleep", lambda _s: None)
        sampler = PeakRssSampler(interval_s=0.05)
        sampler.start()
        rows_seen = 0
        try:
            with _make_streaming_mock_client(synthetic_csv) as client:
                with fetch_with_retry_stream(client, "http://test/csv") as r:
                    assert r is not None
                    assert r.status_code == 200
                    text_stream = io.TextIOWrapper(
                        _IterBytesAdapter(r.iter_bytes(chunk_size=65536)),
                        encoding="utf-8",
                        newline="",
                    )
                    for _row in csv.DictReader(text_stream):
                        rows_seen += 1
        finally:
            sampler.stop()

        assert rows_seen > 0, "no rows parsed from synthetic CSV"
        assert sampler.peak_mb < 200, (
            f"peak RSS {sampler.peak_mb:.1f} MB exceeded 200 MB budget "
            f"with a {FIXTURE_MB} MB fixture (OS-01 regression)"
        )


    def test_embedded_newline_rows_preserved(synthetic_csv, monkeypatch):
        """Validates that newline='' in TextIOWrapper preserves rows whose
        quoted name field contains embedded \\n. Without newline='' the csv
        module would treat the embedded \\n as a row separator and corrupt
        the row count (RESEARCH Pitfall 1)."""
        monkeypatch.setattr("time.sleep", lambda _s: None)
        with _make_streaming_mock_client(synthetic_csv) as client:
            with fetch_with_retry_stream(client, "http://test/csv") as r:
                assert r is not None
                text_stream = io.TextIOWrapper(
                    _IterBytesAdapter(r.iter_bytes(chunk_size=65536)),
                    encoding="utf-8",
                    newline="",
                )
                multiline_rows = 0
                for row in csv.DictReader(text_stream):
                    if "\n" in (row.get("name") or ""):
                        multiline_rows += 1
        assert multiline_rows > 0, (
            "no embedded-newline rows survived parsing; newline='' may be "
            "missing or the synthetic fixture is too small to contain one "
            "(emit one every 1000 rows; FIXTURE_MB=" + str(FIXTURE_MB) + ")"
        )
    ```
