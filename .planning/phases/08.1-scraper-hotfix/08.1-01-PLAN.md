_ROW_PARAM_RE = re.compile(
    r"\['broi_','(\d+)'\]\s*,\s*\['idObj','(\d+)'\]\s*,\s*\['date_izd_','([\d-]+)'\]\s*,\s*\['razdel_','(\d+)'\]"
)


def _parse_issue_rows(html: str) -> list[IssueRow]:
    """Extract issue rows from a broeveList.faces page (razdel_=1 anchors).

    Same row shape Phase 8 parsed on index.faces (verified — broeveList.faces
    encodes [broi_, idObj, date_izd_, razdel_] in the same oamSubmitForm()
    onclick string). razdel_=2 anchors are skipped here; we'll fetch razdel_=2
    explicitly per issue via materiali.faces.
    """
    soup = BeautifulSoup(html, "html.parser")
    rows: list[IssueRow] = []
    seen: set[tuple[int, int]] = set()
    for a in soup.select('a[id^="broi_form:dataTable1:"]'):
        onclick = a.get("onclick", "")
        m = _ROW_PARAM_RE.search(onclick)
        if not m:
            continue
        broi = int(m.group(1))
        id_obj = m.group(2)
        date_izd = date.fromisoformat(m.group(3))
        razdel = int(m.group(4))
        if razdel != 1:
            continue
        year = date_izd.year
        key = (year, broi)
        if key in seen:
            continue
        seen.add(key)
        rows.append(IssueRow(broi=broi, id_obj=id_obj, date_izd=date_izd, razdel=razdel, year=year))
    return rows


_IDMAT_RE = re.compile(r"showMaterialDV\.jsp\?idMat=(\d+)")


def _parse_toc_acts(html: str) -> list[tuple[str, str]]:
    """Parse a per-issue TOC. Returns [(idMat, title), ...].

    See Phase 8 plan 08-01-SUMMARY.md §"Deviations" item 5 for the title-
    extraction story. The materiali.faces TOC has IDENTICAL row shape to the
    Phase 8 index.faces TOC (verified — RESEARCH §Pitfall 9), so this parser
    is byte-identical to Phase 8.
    """
    soup = BeautifulSoup(html, "html.parser")
    acts: list[tuple[str, str]] = []
    seen: set[str] = set()
    for a in soup.select('a[href*="showMaterialDV.jsp"]'):
        href = a.get("href", "")
        m = _IDMAT_RE.search(href)
        if not m:
            continue
        id_mat = m.group(1)
        if id_mat in seen:
            continue
        seen.add(id_mat)
        title = _extract_title_for_anchor(a)
        if not title:
            continue
        acts.append((id_mat, title))
    return acts


def _extract_title_for_anchor(anchor) -> str:
    """Phase 8 mid-task fix — see 08-01-SUMMARY.md §"Deviations" item 5."""
    td = anchor.find_parent("td")
    if td is None:
        return ""
    text = td.get_text("\n", strip=True)
    lines = [ln.strip() for ln in text.split("\n") if ln.strip()]
    strong = td.find("strong")
    body_text = strong.get_text(strip=True) if strong else ""
    anchor_text = anchor.get_text(strip=True)
    page_marker = re.compile(r"^стр\.\s*\d+$")
    candidate_parts: list[str] = []
    for ln in lines:
        if ln == body_text:
            continue
        if ln == anchor_text:
            continue
        if page_marker.match(ln):
            break
        candidate_parts.append(ln)
    return " ".join(candidate_parts)


# ── DB helpers (Phase 8 byte-identical) ───────────────────────────────────


def _upsert_issue(conn, row: IssueRow, source_url: Optional[str] = None) -> str:
    with conn.cursor() as cur:
        cur.execute(
            """
            INSERT INTO dv_issues (issue_number, year, issue_supplement, date, source_url)
            VALUES (%s, %s, %s, %s, %s)
            ON CONFLICT DO NOTHING
            """,
            (row.broi, row.year, row.supplement, row.date_izd, source_url),
        )
        cur.execute(
            "SELECT id FROM dv_issues WHERE issue_number=%s AND year=%s AND issue_supplement=%s",
            (row.broi, row.year, row.supplement),
        )
        rec = cur.fetchone()
    conn.commit()
    return rec[0]


def _upsert_act(
    conn,
    issue_id: str,
    issue_row: IssueRow,
    id_mat: str,
    title: str,
    full_text: str,
    source_url: str,
) -> bool:
    clean_url = strip_jsessionid(source_url)
    act_type = infer_act_type(title)
    with conn.cursor() as cur:
        cur.execute(
            "SELECT id, full_text FROM dv_acts WHERE source_url=%s",
            (clean_url,),
        )
        existing = cur.fetchone()
        if existing:
            existing_id, existing_text = existing
            if existing_text and len(existing_text) > 0:
                conn.commit()
                return False
            cur.execute(
                "UPDATE dv_acts SET full_text=%s WHERE id=%s",
                (full_text, existing_id),
            )
            conn.commit()
            return True
        cur.execute(
            """
            INSERT INTO dv_acts
              (issue_id, issue_number, year, title, act_type, full_text, source_url, razdel)
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
            ON CONFLICT DO NOTHING
            """,
            (issue_id, issue_row.broi, issue_row.year, title, act_type, full_text, clean_url, issue_row.razdel),
        )
    conn.commit()
    return True


# ── View-aliasing guard (D-03) ────────────────────────────────────────────


def _fetch_aliasing_bait(client: httpx.Client) -> set[str]:
    """Fetch the most-recent issue's idMats once at scraper start.

    Used as the live "this is what aliasing looks like" set, in addition to
    the static EXPECTED_ISSUE_2025_100["must_NOT_contain_idmats"] fixture.

    Cheap (one GET); refreshes per scraper run, so the production guard
    auto-tracks whichever issue is currently most-recent.
    """
    html = _fetch_issue_toc(client, MOST_RECENT_ID_OBJ_PROBE, razdel=1)
    return {id_mat for id_mat, _title in _parse_toc_acts(html)}


def _is_aliased(fetched_idmats: set[str], live_bait: set[str]) -> bool:
    """True if the fetched TOC matches the most-recent issue's set exactly.

    Set equality is the cheapest reliable signal: a HISTORICAL issue's idMats
    will never coincidentally equal the most-recent issue's exact set (idMats
    are issued sequentially and don't repeat across issues).

    Empty fetched set is NOT treated as aliased — that's a different failure
    mode (parse failure or genuinely empty issue), handled by the act-count
    log line elsewhere.
    """
    if not fetched_idmats:
        return False
    return fetched_idmats == live_bait


# ── Per-issue handler (with aliasing guard) ───────────────────────────────


def _process_issue(
    client: httpx.Client,
    conn,
    row: IssueRow,
    aliasing_bait: set[str],
) -> int:
    """Fetch BOTH razdel sections for one issue. Returns acts_inserted.

    Source URL stored on dv_issues encodes broi_ + idObj + date_izd_ so that
    --refetch-empty mode can recover idObj from the URL (RESEARCH §Q4 path).
    """
    issue_source = f"{INDEX_URL}?broi_={row.broi}&idObj={row.id_obj}&date_izd_={row.date_izd.isoformat()}"
    issue_id = _upsert_issue(conn, row, source_url=issue_source)
    inserted = 0
    for razdel in (1, 2):
        toc_row = IssueRow(
            broi=row.broi,
            id_obj=row.id_obj,
            date_izd=row.date_izd,
            razdel=razdel,
            year=row.year,
            supplement=row.supplement,
        )
        try:
            toc_html = _fetch_issue_toc(client, row.id_obj, razdel)
        except RuntimeError as exc:
            _log("toc_fetch_failed", id_obj=row.id_obj, razdel=razdel, error=str(exc))
            continue
        toc_acts = _parse_toc_acts(toc_html)
        fetched_idmats = {id_mat for id_mat, _ in toc_acts}
        # D-03 aliasing guard: only flag against the LIVE bait. The most-recent
        # issue is itself allowed to match (it IS the bait) — we exempt it by
        # checking idObj.
        if row.id_obj != MOST_RECENT_ID_OBJ_PROBE and _is_aliased(fetched_idmats, aliasing_bait):
            _log(
                "view_aliasing_detected",
                requested_id_obj=row.id_obj,
                requested_broi=row.broi,
                requested_year=row.year,
                requested_razdel=razdel,
                fetched_idmats=sorted(fetched_idmats),
            )
            continue  # skip insert — do NOT corrupt the DB
        for id_mat, title in toc_acts:
            try:
                body, source_url = _get_act_body(client, id_mat)
            except RuntimeError as exc:
                _log("act_fetch_failed", id_mat=id_mat, error=str(exc))
                continue
            new = _upsert_act(conn, issue_id, toc_row, id_mat, title, body, source_url)
            if new:
                inserted += 1
            _log(
                "act_done",
                id_mat=id_mat,
                razdel=razdel,
                title_prefix=title[:60],
                act_type=infer_act_type(title),
                body_len=len(body),
                inserted=new,
            )
    return inserted


# ── Issue enumeration (date-range walk) ───────────────────────────────────


def _enumerate_issues_for_year(client: httpx.Client, year: int) -> Iterator[IssueRow]:
    """Walk all issues of `year` via broeveList.faces date-filtered POST.

    Pagination terminates when a page returns 0 issue rows. RESEARCH §Q5
    confirms year boundaries don't need special handling; we still per-year-
    walk for clearer logging and finer resume points.
    """
    from_d = date(year, 1, 1)
    to_d = date(year, 12, 31)
    view_state = _enum_initial_view_state(client)
    html, view_state = _enum_first_page(client, view_state, from_d, to_d)
    page_idx = 0
    while True:
        page_idx += 1
        rows = _parse_issue_rows(html)
        if not rows:
            _log("enum_page_empty", year=year, page=page_idx)
            return
        _log(
            "enum_page_done",
            year=year,
            page=page_idx,
            broi_first=rows[0].broi,
            broi_last=rows[-1].broi,
            rows=len(rows),
        )
        for r in rows:
            yield r
        try:
            html, view_state = _enum_next_page(client, view_state, from_d, to_d)
        except ViewExpiredError as exc:
            _log("view_expired_renewing_listing", year=year, page=page_idx, error=str(exc))
            view_state = _enum_initial_view_state(client)
            html, view_state = _enum_first_page(client, view_state, from_d, to_d)


# ── --refetch-empty mode (D-04) ───────────────────────────────────────────


_SOURCE_URL_ID_OBJ_RE = re.compile(r"idObj=(\d+)")


def _parse_id_obj_from_source_url(source_url: str) -> Optional[str]:
    """Recover idObj=N from a stored dv_issues.source_url. None if not present."""
    if not source_url:
        return None
    m = _SOURCE_URL_ID_OBJ_RE.search(source_url)
    return m.group(1) if m else None


def empty_issue_predicate_sql() -> str:
    """The exact SQL used by --refetch-empty mode and by the live-DB test.

    Single source of truth so the test asserts what production uses.
    """
    return (
        "SELECT i.id, i.year, i.issue_number, i.issue_supplement, i.date, i.source_url "
        "FROM dv_issues i "
        "LEFT JOIN dv_acts a ON a.issue_id = i.id "
        "GROUP BY i.id "
        "HAVING COUNT(a.id) = 0 "
        "ORDER BY i.year DESC, i.issue_number DESC"
    )


def _refetch_empty_issues() -> None:
    """Second-pass mode (D-04). Repairs issues with 0 acts."""
    conn = get_pg()
    try:
        with conn.cursor() as cur:
            cur.execute(empty_issue_predicate_sql())
            empties = cur.fetchall()
        _log("refetch_empty_start", count=len(empties))
        if not empties:
            _log("refetch_empty_complete", repaired=0, skipped=0)
            return
        with make_session() as client:
            aliasing_bait = _fetch_aliasing_bait(client)
            _log("aliasing_bait_loaded", count=len(aliasing_bait), id_obj=MOST_RECENT_ID_OBJ_PROBE)
            repaired = 0
            skipped_no_idobj = 0
            for issue_id, year, broi, supplement, date_izd, source_url in empties:
                id_obj = _parse_id_obj_from_source_url(source_url or "")
                if not id_obj:
                    _log("refetch_skip_no_idobj", issue_id=str(issue_id), year=year, broi=broi)
                    skipped_no_idobj += 1
                    continue
                row = IssueRow(
                    broi=broi,
                    id_obj=id_obj,
                    date_izd=date_izd,
                    razdel=1,
                    year=year,
                    supplement=supplement,
                )
                t0 = time.monotonic()
                try:
                    acts = _process_issue(client, conn, row, aliasing_bait)
                except Exception as exc:
                    _log("refetch_issue_error", issue=broi, year=year, error=str(exc))
                    continue
                if acts > 0:
                    repaired += 1
                _log(
                    "refetch_issue_done",
                    issue=broi,
                    year=year,
                    acts=acts,
                    duration_s=round(time.monotonic() - t0, 2),
                )
            _log("refetch_empty_complete", repaired=repaired, skipped=skipped_no_idobj)
    finally:
        conn.close()


# ── Walk loop ─────────────────────────────────────────────────────────────


def _walk(
    issue_filter: Optional[tuple[int, int]] = None,
) -> None:
    """Main walk. Per-year date-filtered enumeration via broeveList.faces.

    If issue_filter=(year, broi), enumerate that year and stop after that issue.
    """
    conn = get_pg()
    try:
        with make_session() as client:
            aliasing_bait = _fetch_aliasing_bait(client)
            _log("aliasing_bait_loaded", count=len(aliasing_bait), id_obj=MOST_RECENT_ID_OBJ_PROBE)
            today = date.today()
            years = list(range(today.year, BACKFILL_FLOOR.year - 1, -1))
            if issue_filter:
                years = [issue_filter[0]]
            total_acts = 0
            seen_issues = 0
            for year in years:
                for row in _enumerate_issues_for_year(client, year):
                    if row.date_izd < BACKFILL_FLOOR:
                        _log("backfill_floor_reached", oldest_date=row.date_izd.isoformat())
                        return
                    if issue_filter and (row.year, row.broi) != issue_filter:
                        continue
                    seen_issues += 1
                    t0 = time.monotonic()
                    try:
                        acts = _process_issue(client, conn, row, aliasing_bait)
                    except Exception as exc:
                        _log("issue_error", issue=row.broi, year=row.year, error=str(exc))
                        continue
                    total_acts += acts
                    _log(
                        "issue_done",
                        issue=row.broi,
                        year=row.year,
                        date=row.date_izd.isoformat(),
                        acts=acts,
                        duration_s=round(time.monotonic() - t0, 2),
                    )
                    if issue_filter and (row.year, row.broi) == issue_filter:
                        _log("smoke_complete", total_acts=total_acts)
                        return
            _log("walk_complete", years=years, issues=seen_issues, total_acts=total_acts)
    finally:
        conn.close()


def _parse_issue_arg(s: str) -> tuple[int, int]:
    year_s, broi_s = s.split("/")
    return int(year_s), int(broi_s)


def main() -> None:
    parser = argparse.ArgumentParser(description="Scrape dv.parliament.bg into dv_issues + dv_acts")
    parser.add_argument(
        "--issue",
        metavar="YYYY/NN",
        help="Smoke-test mode: scrape only this single issue (e.g. 2025/100)",
    )
    parser.add_argument(
        "--refetch-empty",
        action="store_true",
        help="Second-pass mode: refetch acts for any dv_issues row with 0 acts (D-04)",
    )
    args = parser.parse_args()

    if "DATABASE_URL" not in os.environ:
        print("Error: DATABASE_URL not set in environment", file=sys.stderr)
        sys.exit(1)

    if args.refetch_empty:
        _log("refetch_empty_mode_start")
        _refetch_empty_issues()
        return

    issue_filter: Optional[tuple[int, int]] = None
    if args.issue:
        issue_filter = _parse_issue_arg(args.issue)
        _log("smoke_start", issue=args.issue)
    else:
        _log("backfill_start", floor=BACKFILL_FLOOR.isoformat())

    _walk(issue_filter=issue_filter)


if __name__ == "__main__":
    main()
