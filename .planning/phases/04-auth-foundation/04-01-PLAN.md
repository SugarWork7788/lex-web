```sql
-- Phase 4 — Auth foundation: user_profiles + RLS + trigger
-- Idempotent — safe to re-run.
-- Per D-05 + D-06 (no locale column) + RESEARCH §"Recommended db/auth_schema.sql"
--   (SECURITY DEFINER hardened with SET search_path = public).

CREATE TABLE IF NOT EXISTS user_profiles (
  id uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  display_name text NOT NULL,
  created_at timestamptz DEFAULT now()
);

ALTER TABLE user_profiles ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS "users can read own profile" ON user_profiles;
CREATE POLICY "users can read own profile" ON user_profiles
  FOR SELECT USING (auth.uid() = id);

DROP POLICY IF EXISTS "users can update own profile" ON user_profiles;
CREATE POLICY "users can update own profile" ON user_profiles
  FOR UPDATE USING (auth.uid() = id);

CREATE OR REPLACE FUNCTION handle_new_user() RETURNS trigger
  LANGUAGE plpgsql SECURITY DEFINER SET search_path = public
  AS $$
  BEGIN
    INSERT INTO public.user_profiles (id, display_name)
    VALUES (
      NEW.id,
      COALESCE(
        NEW.raw_user_meta_data->>'display_name',  -- email signup (D-03)
        NEW.raw_user_meta_data->>'full_name',     -- Google OAuth (D-03)
        split_part(NEW.email, '@', 1)             -- safety net
      )
    );
    RETURN NEW;
  END;
  $$;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();
```

```typescript
import { readFileSync } from "node:fs";
import { Client } from "pg";

function maskUrl(url: string): string {
  return url.replace(/(:[^:@/]+@)/, ":***@");
}

async function main() {
  const dbUrl = process.env.DATABASE_URL;
  if (!dbUrl) {
    console.error("Error: DATABASE_URL not set in environment.");
    console.error("Hint: source ../lex-brain/.env or supply DATABASE_URL inline.");
    process.exit(1);
  }

  const sqlPath = `${process.cwd()}/db/auth_schema.sql`;
  const sql = readFileSync(sqlPath, "utf-8");

  console.log(`Applying auth schema migration from ${sqlPath} to ${maskUrl(dbUrl)}…`);

  const client = new Client({
    connectionString: dbUrl,
    ssl: { rejectUnauthorized: false },
  });
  await client.connect();

  try {
    await client.query("BEGIN");
    await client.query(sql);
    await client.query("COMMIT");
    console.log("Schema applied successfully.");
  } catch (err) {
    await client.query("ROLLBACK");
    console.error("Schema apply failed:", err);
    await client.end();
    process.exit(1);
  }

  let allOk = true;
  const probes: { name: string; query: string; expect: number }[] = [
    {
      name: "user_profiles table exists",
      query: `SELECT table_name FROM information_schema.tables WHERE table_schema='public' AND table_name='user_profiles'`,
      expect: 1,
    },
    {
      name: "RLS enabled on user_profiles",
      query: `SELECT relname FROM pg_class WHERE relname='user_profiles' AND relrowsecurity=true`,
      expect: 1,
    },
    {
      name: "user_profiles has both RLS policies (read + update)",
      query: `SELECT policyname FROM pg_policies WHERE tablename='user_profiles'`,
      expect: 2,
    },
    {
      name: "handle_new_user is SECURITY DEFINER with search_path=public (hardened)",
      query: `SELECT proname FROM pg_proc WHERE proname='handle_new_user' AND prosecdef=true AND proconfig::text LIKE '%search_path=public%'`,
      expect: 1,
    },
    {
      name: "on_auth_user_created trigger registered on auth.users",
      query: `SELECT tgname FROM pg_trigger WHERE tgname='on_auth_user_created' AND tgrelid='auth.users'::regclass`,
      expect: 1,
    },
  ];

  for (const probe of probes) {
    const res = await client.query(probe.query);
    if (res.rowCount === probe.expect) {
      console.log(`OK: ${probe.name} (${res.rowCount}/${probe.expect})`);
    } else {
      console.error(`MISSING: ${probe.name} (got ${res.rowCount}, expected ${probe.expect})`);
      allOk = false;
    }
  }

  await client.end();
  process.exit(allOk ? 0 : 1);
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser/API → Supabase REST + Auth | Future requests carry `auth.uid()` derived from JWT; RLS must enforce per-row isolation |
| Authenticated session → user_profiles | SELECT/UPDATE policies must restrict to `auth.uid() = id` |
| `auth.users` INSERT (Supabase-managed) → public.user_profiles INSERT (our trigger) | SECURITY DEFINER crosses the auth schema → public schema boundary; search_path injection is the canonical risk |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Elevation of Privilege | RLS bypass via direct PostgREST access (`SELECT * FROM user_profiles` from another user's session) | mitigate | Both RLS policies enforce `auth.uid() = id`; `ENABLE ROW LEVEL SECURITY` is on; probe 2 + probe 3 in apply-auth-schema.ts verifies both. No `BYPASSRLS` granted. |
| T-04-02 | Elevation of Privilege | `SECURITY DEFINER` function `handle_new_user` exploited via `search_path` injection (attacker creates a malicious `public.user_profiles` view in their search_path before triggering an auth.users INSERT) | mitigate | `SET search_path = public` pinned in function definition (RESEARCH-corrected — exceeds CONTEXT D-05's spec). Probe 4 asserts `proconfig` contains `search_path=public`. |
| T-04-03 | Tampering | App code path inserting into user_profiles racing with the trigger (double-row, TOCTOU on first sign-in) | mitigate | D-05 LOCK: app code MUST NEVER insert into user_profiles. Only the trigger writes. Plans 04-02 and 04-03 have zero `INSERT INTO user_profiles` or `from('user_profiles').insert(...)` (verifiable via grep). |
| T-04-04 | Denial of Service | Trigger raises an exception → blocks `auth.users` insert → user can't sign up | accept | The COALESCE-with-split_part fallback ensures `display_name` is never NULL even with empty `raw_user_meta_data`. The only failure modes are catastrophic (DB-level) and would block all signups anyway. |
| T-04-05 | Information Disclosure | `display_name` exposed via leaked SELECT to unauthorized session | mitigate | SELECT policy = `auth.uid() = id` — only self. Anonymous role has no RLS-granted access (default deny). |
| T-04-06 | Tampering | Migration applied to wrong DB (e.g. user's local dev DB instead of prod) | accept | `maskUrl()` logs the masked URL pre-apply so operator can sanity-check the host; the BLOCKING checkpoint requires the operator to read the log. |
</threat_model>
